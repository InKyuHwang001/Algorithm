## 스택의 간단한 구현

---
### 함수
```python
def isStackFull():
    global SIZE,stack,top
    if(top>=SIZE-1):
        return True
    else:
        return False
def push(data):
    global SIZE,stack,top
    if(isStackFull()):
        print('스택 꽉참')
        return
    top+=1
    stack[top]=data
def isStackEmpty():
    global SIZE,stack,top
    if (top==-1):
        return True
    else:
        return False
def pop():
    global SIZE,stack,top
    if (isStackEmpty()):
        print('스택 텅~')
        return None
    data=stack[top]
    stack[top]=None
    top-=1
    return data
```
