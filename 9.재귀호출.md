```python
def factorial(num):
    if num<=1:
        print('1반환')
        return 1
    print("%d*%d! 호출"%(num, num-1))
    retVal=factorial(num-1)
    print("%d*%d!(=%d) 반환"%(num, num-1,retVal))
    return num*retVal
print('\n5!',factorial(5)
```

    if n==0:

def gugu(dan, num):
    print("%d*%d=%d"%(dan,num,dan*num))
    if num<19:
        gugu(dan,num+1)
for dan in range(2,20):
    print("## %d단 ##"%dan)
    gugu(dan,1)

### N제곱 계산하기

```python
tab=""
def pow(x,n):
    global tab
    tab+=" "
    if n ==0:
        return 1
    print(tab+"%d*%d^(%d-%d)"%(x,x,n,1))
    return x*pow(x,n-1)
print('2^4')
print('답-->',pow(2,4))
```

### 배열의 합 계산하기

```python
import random
def arySum(arr,n):
    if n<=0:
        return arr[0]
    return arySum(arr, n-1)+arr[n]
ary=[random.randint(0,255) for _ in range(random.randint(10,20))]
print(ary)
print('배열 합계-->',arySum(ary,len(ary)-1))
```
